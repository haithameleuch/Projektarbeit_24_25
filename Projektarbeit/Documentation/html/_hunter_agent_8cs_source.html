188 public override void OnActionReceived(ActionBuffers actions)
189 {
190 if (!target) return;
191
192 var moveInput = actions.ContinuousActions[0]; // Forward/backward
193 var turnInput = actions.ContinuousActions[1]; // Left/right turn
194
195 // Move forward/backward
196 var forwardMovement = transform.forward * moveInput * movementSpeed * Time.deltaTime;
197 _rb.MovePosition(_rb.position + forwardMovement);
198
199 // Rotate
200 transform.Rotate(Vector3.up, turnInput * 180f * Time.deltaTime);
201
202 // Calculate distance-based progress reward
203 var currentDistance = Vector3.Distance(transform.localPosition, target.transform.localPosition);
204 var distanceDelta = _prevDistance - currentDistance;
205 AddReward(distanceDelta * 0.01f);
206 _prevDistance = currentDistance;
207
208 // Reward for facing the target
209 var toTarget = (target.transform.localPosition - transform.localPosition).normalized;
210 var facingDot = Vector3.Dot(transform.forward, toTarget); // 1 = facing directly at target
211 AddReward(facingDot * 0.005f);
212
213 // Step penalty to encourage faster completion
214 AddReward(-0.001f);
215
216 // Check if the agent is stuck (not moving)
217 if (Vector3.Distance(transform.localPosition, _lastPosition) < 0.01f)
218 {
219 _stuckTimer += Time.deltaTime;
220 if (_stuckTimer > 2f)
221 {
222 AddReward(-5f);
223 EndEpisode();
224 }
225 }
226 else
227 {
228 _stuckTimer = 0f;
229 }
230
231 _lastPosition = transform.localPosition;
