118 public Circle GetCircumcircle()
119 {
120 // Circum center
121 var d = 2 * (Points[0].X * (Points[1].Y - Points[2].Y) + Points[1].X * (Points[2].Y - Points[0].Y) + Points[2].X * (Points[0].Y - Points[1].Y));
122 var ux = (
123 (Points[0].X * Points[0].X + Points[0].Y * Points[0].Y) * (Points[1].Y - Points[2].Y)
124 + (Points[1].X * Points[1].X + Points[1].Y * Points[1].Y) * (Points[2].Y - Points[0].Y)
125 + (Points[2].X * Points[2].X + Points[2].Y * Points[2].Y) * (Points[0].Y - Points[1].Y)
126 ) / d;
127 var uy = (
128 (Points[0].X * Points[0].X + Points[0].Y * Points[0].Y) * (Points[2].X - Points[1].X)
129 + (Points[1].X * Points[1].X + Points[1].Y * Points[1].Y) * (Points[0].X - Points[2].X)
130 + (Points[2].X * Points[2].X + Points[2].Y * Points[2].Y) * (Points[1].X - Points[0].X)
131 ) / d;
132 var center = new Point(ux, uy);
133 var radius = Mathf.Sqrt((Points[0].X-center.X)* (Points[0].X - center.X) + (Points[0].Y-center.Y)* (Points[0].Y - center.Y));
134 return new Circle(center,radius);

143 public static Point GetIntersection(Edge e1, Edge e2)
144 {
145 var isVert1 = false;
146 var isVert2 = false;
147
148 var m1 = 0f;
149 var b1 = 0f;
150 var m2 = 0f;
151 var b2 = 0f;
152
153 // Try to get function parameters of e_1
154 try
155 {
156 m1 = (e1.B.Y - e1.A.Y) / (e1.B.X - e1.A.X);
157 b1 = (e1.A.Y - m1 * e1.A.X);
158 }
159 catch (DivideByZeroException)
160 {
161 // Catch if edge is vertical (division by 0 for m)
162 isVert1 = true;
163 }
164
165 // Try to get function parameters of e_2
166 try
167 {
168 m2 = (e2.B.Y - e2.A.Y) / (e2.B.X - e2.A.X);
169 b2 = (e2.A.Y - m2 * e2.A.X);
170 }
171 catch (DivideByZeroException)
172 {
173 // Catch if edge is vertical (division by 0 for m)
174 isVert2 = true;
175 }
176
177 // This should be impossible
178 if (isVert1 && isVert2)
179 {
180 return null;
181 }
182
183 if (isVert1)
184 {
185 // Return point with x of the vertical line and y with the x plugged in the equation of the none vertical line
186 return new Point(e1.A.X, m2 * e1.A.X + b2);
187 }
188 else if (isVert2)
189 {
190 // Return point with x of the vertical line and y with the x plugged in the equation of the none vertical line
191 return new Point(e2.A.X, m1 * e2.A.X + b1);
192 }
193 else
194 {
195 // Set both equations equal and rearranged to calculate x and use x to calculate y
196 var x = (b2 - b1) / (m1 - m2);
197 var y = m1 * x + b1;
198
199 return new Point(x, y);
200 }

358 bool OnSegment(Point p, Point q, Point r)
359 {
360 return q.X <= Math.Max(p.X, r.X) && q.X >= Math.Min(p.X, r.X) &&
361 q.Y <= Math.Max(p.Y, r.Y) && q.Y >= Math.Min(p.Y, r.Y);
362 }
363
364 var o1 = Orientation(p1, p2, q1);
365 var o2 = Orientation(p1, p2, q2);
366 var o3 = Orientation(q1, q2, p1);
367 var o4 = Orientation(q1, q2, p2);
368
369 // General case
370 if (o1 * o2 < 0 && o3 * o4 < 0)
371 return true;
372
373 // Special cases (colinear points)
374 if (o1 == 0 && OnSegment(p1, q1, p2)) return true;
375 if (o2 == 0 && OnSegment(p1, q2, p2)) return true;
376 if (o3 == 0 && OnSegment(q1, p1, q2)) return true;
377 if (o4 == 0 && OnSegment(q1, p2, q2)) return true;
