177 private void Update()
178 {
179 if (_player is null) return;
180
181 // Game over logic
182 // Check if the player is dead
183 if (_player.GetComponent<Stats.Stats>().GetCurStats(0) < 1)
184 {
185 // Pause the game
186 DisableGameplay();

188 // Lock the cursor in the game view
189 HidePanel();
190 CloseAllUIExcept(null);
191
192 if (!_gameOverShown)
193 {
194 if (gameOverLevelText is null && gameOver is not null)
195 {
196 var t = gameOver.transform.Find("Level");
197 if (t is not null) gameOverLevelText = t.GetComponent<TMP_Text>();
198 }
199
200 if (gameOverLevelText is not null)
201 gameOverLevelText.text = $"Level: {SaveSystemManager.GetLevel()}";
202
203 _gameOverShown = true;
204 }
205
206 gameOver.gameObject.SetActive(true);
207 GameInputManager.Instance.MouseLocked(false);
208
209 // Pause the game time
210 Time.timeScale = 0;
211 return;
212 }
213
214 if (!_isPauseVisible && !_isInvVisible) // only resume if not paused or in inventory
215 {
216 _gameOverShown = false;
217
218 // Resume the game time
219 Time.timeScale = 1;
220
221 // Unlock the cursor
222 EnableGameplay();
223 }
224
225 //Check weather "P" is pressed to toggle the pause menu
226 if (Input.GetKeyDown(KeyCode.P) && !_isCombatLocked)
227 {
228 if (_isPauseVisible)
229 {
230 //_player.SetActive(true);
231 pause.gameObject.SetActive(false);
232
233 //Lock Cursor in the game view
234 GameInputManager.Instance.MouseLocked(true);
235
236 //Resume time to normal value
237 Time.timeScale = 1;
238
239 EnableGameplay();
240
241 _isPauseVisible=false;
242 }
243 else
244 {
245 // Only one overlay at a time
246 CloseAllUIExcept("pause");
247
248 // Set level text on the pause menu
249 if (pauseLevelText is null && pause is not null)
250 {
251 var t = pause.transform.Find("Level");
252 if (t is not null) pauseLevelText = t.GetComponent<TMP_Text>();
253 }
254 if (pauseLevelText is not null)
255 pauseLevelText.text = $"Level: {SaveSystemManager.GetLevel()}";
256
257 DisableGameplay();
258 HidePanel();
259
260 pause.gameObject.SetActive(true);
261
262 //Make the cursor moveable within the game window
263 GameInputManager.Instance.MouseLocked(false);
264
265 //Pause the game
266 Time.timeScale = 0;
267 _isPauseVisible=true;
268 }
269 }
270
271 if (Input.GetKeyDown(KeyCode.I))
272 {
273 if (_isPauseVisible) return;
274 if (!_isInvVisible)
275 {
276 HidePanel();
277
278 DisableGameplay();
279
280 CloseAllUIExcept("inventory");
281
282 _itemUI.transform.parent.gameObject.SetActive(true);
283
284 //Make the cursor moveable within the game window
285 GameInputManager.Instance.MouseLocked(false);
286
287 //Pause the game
288 Time.timeScale = 0;
289
290 _isInvVisible = true;
291 }
292 else
293 {
294 EnableGameplay();
295
296 _itemUI.transform.parent.gameObject.SetActive(false);
297
298 //Make the cursor unmoveable within the game window
299 GameInputManager.Instance.MouseLocked(true);
300
301 //Pause the game
302 Time.timeScale = 1;
303
304 _isInvVisible = false;
305 }
306 }
307
308 if (Input.GetKeyDown(KeyCode.M))
309 {
310 if (miniMapPanel is null) return;
311
312 // don't allow the map while game over or pause
313 if ((gameOver is not null && gameOver.gameObject.activeSelf) || _isPauseVisible) return;
314
315 var isActive = miniMapPanel.activeSelf;
316
317 if (!isActive)

319 // only one overlay at a time
320 CloseAllUIExcept("map");
321
322 // Ensure gameplay is running (the map does not pause)
323 ResumeGameplayIfNotGameOver();

329 miniMapPanel.SetActive(false);
