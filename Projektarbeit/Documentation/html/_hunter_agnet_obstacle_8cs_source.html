176 public override void OnActionReceived(ActionBuffers actions)
177 {
178 if (!target) return;
179
180 var moveInput = Mathf.Clamp(actions.ContinuousActions[0], 0f, 1f); // Forward/backward input
181 var turnInput = Mathf.Clamp(actions.ContinuousActions[1], -1f, 1f); // Left/right rotation input
182
183 // Compute movement vector based on input and speed, then move the agent
184 var movement = transform.forward * moveInput * movementSpeed * Time.deltaTime;
185 _rb.MovePosition(_rb.position + movement);
186
187 // Rotate the agent smoothly around Y-axis
188 transform.Rotate(Vector3.up, turnInput * 300f * Time.deltaTime);
189
190 // Compute distance to the target and progress since last step
191 var currentDistance = Vector3.Distance(transform.localPosition, target.transform.localPosition);
192 var distanceDelta = _prevDistance - currentDistance;
193
194 // Scale progress reward based on current distance
195 var distanceScale = Mathf.Max(1f, currentDistance / 2f);
196 var progressReward = distanceDelta * distanceScale * 2f;
197
198 // Compute alignment of agent's velocity with direction to target
199 var directionToTarget = (target.transform.position - transform.position).normalized;
200 var movementAlignment = Vector3.Dot(_rb.linearVelocity.normalized, directionToTarget);
201 var alignmentReward = movementAlignment * 0.2f;
202
203 // Compute reward for facing the target
204 var facingDot = Vector3.Dot(transform.forward, directionToTarget);
205 var facingReward = Mathf.Pow(facingDot, 2) * 0.4f;
206
207 // Add combined reward for this step
208 var totalReward = progressReward + alignmentReward + facingReward;
209 AddReward(totalReward);
210
211 _prevDistance = currentDistance;
212
213 // Detect if the agent is stuck and penalize if stuck for too long
214 if (Vector3.Distance(transform.localPosition, _lastPosition) < StuckThreshold)
215 {
216 _stuckTimer += Time.deltaTime;
217 if (_stuckTimer > StuckTimeLimit)
218 {
219 AddReward(-0.5f);
220 return;
221 }
222 }
223 else
224 {
225 _stuckTimer = 0f;
226 }
227
228 _lastPosition = transform.localPosition;
229 }
