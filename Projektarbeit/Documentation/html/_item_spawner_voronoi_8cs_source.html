65 Random.InitState(SaveSystemManager.GetSeed());
66
67 var collectibleIndex = 0;
68
69 // random initial counts per item room
70 var plannedCounts = new List<int>(_rooms.Count);
71 for (var r = 0; r < _rooms.Count; r++)
72 plannedCounts.Add(Random.Range(1, 5)); // 1–4 items per room
73
74 // make sure total slots >= number of must items (from Distributor)
75 var mustTotal = _itemsDistributor.MustCount;
76 var totalSlots = 0;
77 for (var i = 0; i < plannedCounts.Count; i++) totalSlots += plannedCounts[i];
78
79 // increase slots per room if necessary
80 var roomBumpIdx = 0;
81 while (totalSlots < mustTotal && _rooms.Count > 0)
82 {
83 plannedCounts[roomBumpIdx] += 1;
84 totalSlots++;
85 roomBumpIdx = (roomBumpIdx + 1) % _rooms.Count;
86 }
87
88 for (var roomIdx = 0; roomIdx < _rooms.Count; roomIdx++)
89 {
90 var room = _rooms[roomIdx];
91 var itemCount = plannedCounts[roomIdx];
92 var radius = room.GetIncircleRadius();
93
94 for (var i = 0; i < itemCount; i++)
95 {
96 var idx = collectibleIndex++;
97 var itemInstance = _itemsDistributor.GetRandomElementIncludingMust();
98
99 // Place items in a circular pattern
100 var angle = i * (360f / itemCount);
101 var distanceFromCenter = Mathf.Min(radius * 0.6f, 3f); // keep items within safe bounds
102
103 var xOffset = Mathf.Cos(angle * Mathf.Deg2Rad) * distanceFromCenter;
104 var zOffset = Mathf.Sin(angle * Mathf.Deg2Rad) * distanceFromCenter;
105
106 var spawnPos = new Vector3(room.Center.X + xOffset, 0.5f, room.Center.Y + zOffset);
107 Quaternion rotation = Quaternion.Euler(0, Random.Range(0f, 360f), 0);
108
109 var spawnedItem = Object.Instantiate(itemInstance.itemData._model, spawnPos, rotation, _parent);
110
111 var auraTransform = spawnedItem.transform.Find("SphereAura/Aura");
112 if (auraTransform is not null)
113 {
114 var renderer = auraTransform.GetComponent<MeshRenderer>();
115
116 var material = renderer.material;
117 material.SetColor("_ColorA", itemInstance.itemData.GetRarityColor());
118 material.SetColor("_ColorB", itemInstance.itemData.GetRarityColor());
119 }
120
121 var collectible = spawnedItem.GetComponent<CollectibleItem>();
122 if (collectible is not null)
123 {
124 collectible.saveIndex = idx;
125 collectible.Initialize(itemInstance.itemData);
126 collectible.amount = itemInstance.itemQuantity;
127 }
128
129 if (!SaveSystemManager.IsCollectibleActive(idx))
130 {
131 spawnedItem.SetActive(false);
132 continue;
133 }
134
135 spawnedItem.SetActive(true);
136 }
