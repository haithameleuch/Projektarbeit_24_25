309 private void RandomizeSizeAndPoints()
310 {
311 var level = Mathf.Max(1, SaveSystemManager.GetLevel());
312 var rnd = new System.Random(_seed ^ 0xA2C2A);
313
314 var t = Mathf.Clamp01((level - 1) / 9f);
315
316 // Size 40..80 + small Seed-Noise (±5)
317 var baseSize = Mathf.Lerp(40f, 80f, t);
318 var sizeNoise = (float)(rnd.NextDouble() * 2 - 1) * 5f; // ±5
319 size = Mathf.Clamp(Mathf.Round(baseSize + sizeNoise), 40f, 80f);
320
321 // numPoints proportional with size 10..20 + Jitter (±2)
322 var prop = (size - 40f) / 40f; // 0..1
323 var basePoints = Mathf.RoundToInt(Mathf.Lerp(10f, 20f, prop));
324 var jitter = rnd.Next(-2, 3);
325 numPoints = Mathf.Clamp(basePoints + jitter, 10, 20);
326
327 Debug.Log($"[Voronoi] Level{level} -> size={size}, numPoints={numPoints}");
328 }

385 private HashSet<int> GetNeighbors(int pointID, Room room)
386 {
387 var target = _debugPoints[pointID];
388 var neighbors = new HashSet<int>();
389
390 foreach (var t in _debugTriangles)
391 {
392 for (var i = 0; i < 3; i++)
393 {
394 if (!Point.Equals(t.Points[i], target)) continue;
395
396 var n1 = (i + 1) % 3;
397 var n2 = (i + 2) % 3;
398
399 var idx1 = _debugPoints.FindIndex(p => Point.Equals(p, t.Points[n1]));
400 var idx2 = _debugPoints.FindIndex(p => Point.Equals(p, t.Points[n2]));
401
402 if (idx1 != -1) neighbors.Add(idx1);
403 if (idx2 != -1) neighbors.Add(idx2);
404
405 break;
406 }
407 }
408
409 return neighbors;

573 private void CreateWall(Vector3 start, Vector3 end, int edgeId)
574 {
575 // Get all rooms that are touching this edge
576 var roomsWithEdge = _dungeonGraph.Rooms.Where(r => r.Walls.Contains(edgeId));
577
578 // Prefab width and number of segments
579 var widthOfPrefab = 2;
580 var numberOfSegments = Mathf.Max(1, (int)(Vector3.Distance(start, end) / widthOfPrefab));
581 var scaleOfSegment = (Vector3.Distance(start, end) / widthOfPrefab) / numberOfSegments;
582 var step = (end - start) / numberOfSegments;
583
584 // Detect whether this entire edge is an exterior wall
585 var isBoundary =
586 (Mathf.Approximately(start.z, 0f) && Mathf.Approximately(end.z, 0f)) || // lower edge
587 (Mathf.Approximately(start.z, size) && Mathf.Approximately(end.z, size)) || // top edge
588 (Mathf.Approximately(start.x, 0f) && Mathf.Approximately(end.x, 0f)) || // left edge
589 (Mathf.Approximately(start.x, size) && Mathf.Approximately(end.x, size)); // right edge
590
591 var longEnoughForDoor = Vector3.Distance(start, end) >= minDoorEdgeLength;
592
593 // Which segment is in the middle
594 var midIndex = numberOfSegments / 2;
595
596 for (var i = 0; i < numberOfSegments; i++)
597 {
598 var segStart = start + step * i;
599 var segEnd = start + step * (i + 1);
600
601 // Determine whether this edge must contain a door
602 var forceBossDoor = roomsWithEdge.Any(r => r.Type == RoomType.Boss);
603
604 var forcePathDoor = false;
605 if (roomsWithEdge.Count() == 2)
606 {
607 var a = roomsWithEdge.ElementAt(0).ID;
608 var b = roomsWithEdge.ElementAt(1).ID;
609 forcePathDoor = _forcedDoorPairs.Contains(GetPairKey(a, b));
610 }
611
612 var forceDoor = forceBossDoor || forcePathDoor;
613 var forbidBreakable = roomsWithEdge.Any(r => r.Type == RoomType.Boss);
614 var placeDoor = forceDoor || (_rng.NextDouble() < DoorProbability);
615
616 var isMiddle = i == midIndex;
617 var candidate = !isBoundary && isMiddle && longEnoughForDoor;
618
619 if (candidate && (placeDoor || forceDoor))
620 {
621 // Place a door prefab in the middle
622 var mid = (segStart + segEnd) * 0.5f;
623 var rot = Quaternion.FromToRotation(Vector3.right, segEnd - segStart);
624
625 var doorObj = Instantiate(door, mid, rot, transform);
626 doorObj.transform.localScale = new Vector3(scaleOfSegment, 1f, 1f);
627
628 if (forceBossDoor)
629 {
630 SetupBossDoor(doorObj, roomsWithEdge.ToList());
631 }
632
633 // Add a door to each room
634 foreach (var room in roomsWithEdge)
635 room.Doors.Add(edgeId);
636
637 _dungeonGraph.IDDoorDict[edgeId] = doorObj;
638 }
639 else if (candidate && !forbidBreakable)
640 {
641 // Place a breakable wall instead of a door
642
643 var localIdx = _breakableWallCounter++;
644
645 var activeList = SaveSystemManager.GetDestroyableWallsActive();
646 while (activeList.Count <= localIdx)
647 activeList.Add(true);
648
649 var healthList = SaveSystemManager.GetDestroyableWallsHealth();
650 while (healthList.Count <= localIdx)
651 healthList.Add(Random.Range(1, 6));
652
653 if (activeList[localIdx])
654 {
655 var mid = (segStart + segEnd) * 0.5f;
656 var rot = Quaternion.FromToRotation(Vector3.right, segEnd - segStart);
657 var brkObj = Instantiate(destroyableWall, mid, rot, transform);
658 brkObj.transform.localScale = new Vector3(scaleOfSegment, 1f, 1f);
659
660 var interaction = brkObj.GetComponent<DestroyableWallInteraction>();
661 if (interaction != null)
662 {
663 interaction.edgeID = localIdx;
664 interaction.InitializeFromSave();
665 }
666 }
667 // ---- DESTROYABLE WALL ----
668 }
669 else
670 {
671 // Place a regular wall segment
672 CreateWallSegment(segStart, segEnd, scaleOfSegment);
673 }
674 }
675 }

683 private void SetupBossDoor(GameObject doorObj, List<Room> roomsWithEdge)
684 {
685 // Ensure the door has an OpenDoor component, add one if missing
686 var openDoor = doorObj.GetComponent<OpenDoor>() ??
687 doorObj.GetComponentInChildren<OpenDoor>(true) ??
688 doorObj.AddComponent<OpenDoor>();
689 openDoor.isBossDoor = true;
690
691 // Find the child object Wall_Entrance
692 var entrance = doorObj.transform.Find("Wall_Entrance");
693 if (entrance == null) return;
694
695 // Find objects fog wall and door panel
696 var fog = entrance.Find("Room_Door_Fog");
697 var doorPanel = entrance.Find("Door");
698
699 // Identify the boss room and the adjacent room (otherRoom)
700 var roomList = roomsWithEdge.ToList();
701 var bossRoom = roomList.FirstOrDefault(r => r.Type == RoomType.Boss);
702 var otherRoom = roomList.FirstOrDefault(r => r != bossRoom);
703
704 if (bossRoom != null && otherRoom != null)
705 {
706 // Determine the direction from boss room to the other room
707 var dir = (otherRoom.Center.ToVector3() - bossRoom.Center.ToVector3()).normalized;
708
709 // Check if the entrance is facing the wrong direction
710 var alignment = Vector3.Dot(dir, entrance.forward);
711
712 // Rotate 180° if the entrance is facing the wrong direction
713 if (alignment > 0f)
714 entrance.Rotate(0, 180f, 0);
715 }
716
717 // Enable fog wall
718 if (fog != null)
719 fog.gameObject.SetActive(true);
720
721 // Make the door visually narrower
722 if (doorPanel != null)
723 {
724 var scale = doorPanel.localScale;
725 scale.z = 0.75f;
726 doorPanel.localScale = scale;
727 }
728 }

806 private void ComputeForcedDoorPairs()
807 {
808 _forcedDoorPairs.Clear();
809 ForcedItemRoomId = -1;
810
811 var start = _dungeonGraph.GetStartRoom();
812 var itemRooms = _dungeonGraph.GetAllItemRooms();
813 if (start == null || itemRooms == null || itemRooms.Count == 0) return;
814
815 List<Room> bestPath = null;
816 Room bestTarget = null;
817
818 foreach (var item in itemRooms)
819 {
820 var path = _dungeonGraph.FindShortestPathWithoutBossRoom(start, item);
821 if (path == null) continue;
822
823 if (bestPath == null || path.Count < bestPath.Count)
824 {
825 bestPath = path;
826 bestTarget = item;
827 }
828 }
829
830 // Fallback
831 if (bestPath == null)
832 {
833 var firstItem = itemRooms.FirstOrDefault();
834 if (firstItem == null) return;
835
836 var fallback = _dungeonGraph.FindShortestPath(start, firstItem);
837 if (fallback == null) return;
838
839 for (int i = 0; i < fallback.Count - 1; i++)
840 _forcedDoorPairs.Add(GetPairKey(fallback[i].ID, fallback[i + 1].ID));
841
842 ForcedItemRoomId = firstItem.ID;
843 return;
844 }
845
846 for (var i = 0; i < bestPath.Count - 1; i++)
847 _forcedDoorPairs.Add(GetPairKey(bestPath[i].ID, bestPath[i + 1].ID));
848
849 ForcedItemRoomId = bestTarget?.ID ?? bestPath.Last().ID;
850 }

891 private List<Point> GeneratePoints(int count, float radius, float dungeonSize)
892 {
893 var points = new List<Point>();
894 var maxAttempts = 1000;
895
896 while (points.Count < count && maxAttempts > 0)
897 {
898 var margin = 2f;
899 var x = margin + (float)_rng.NextDouble() * (dungeonSize - 2 * margin);
900 var y = margin + (float)_rng.NextDouble() * (dungeonSize - 2 * margin);
901 var newPoint = new Point(x, y);
902 var isValid = true;
903
904 foreach (var existing in points)
905 {
906 if (Point.GetDistance(existing, newPoint) < radius)
907 {
908 isValid = false;
909 break;
910 }
911 }
912
913 if (isValid)
914 {
915 points.Add(newPoint);
916 }
917
918 maxAttempts--;
919 }
920
921 return points;
922 }

929 private List<Triangle> BowyerWatson(List<Point> points)
930 {
931 // List of triangles forming the Delaunay-Triangulation
932 var triangulation = new List<Triangle>();
933
934 // A triangle that is big enough so all points fall within its bound
935 var superTriangle = new Triangle(new Point(-size,-size/2), new Point(size/2,size*2.5f), new Point(2*size,-size/2));
936 triangulation.Add(superTriangle);
937
938 // ONLY FOR DEBUGGING
939 _debugSuperTriangle = superTriangle;
940 // ONLY FOR DEBUGGING
941
942 // Add each point after point
943 foreach (var point in points)
944 {

946 var badTriangles = new List<Triangle>();
947
948 // Find all invalid triangles in the triangulation
949 foreach (var triangle in triangulation)
950 {
951 if (triangle.IsInCircumcircle(point))
952 {
953 badTriangles.Add(triangle);
954 }
955 }
956
957 var polygon = new List<Edge>();
958
959 // Find boundary polygon for generation of new triangles
960 foreach (var triangle in badTriangles)
961 {
962 foreach (var edge in triangle.Edges)
963 {
964 var count = 0;
965 foreach (var other in badTriangles)
966 {
967 if (Edge.Equals(edge, other.Edges[0]))
968 {
969 count++;
970 }
971 else if (Edge.Equals(edge, other.Edges[1]))
972 {
973 count++;
974 }
975 else if (Edge.Equals(edge, other.Edges[2]))
976 {
977 count++;
978 }
979 }
980
981 // Edge is only in one triangle -> at the border
982 if (count == 1)
983 {
984 polygon.Add(edge);
985 }
986 }
987 }
988
989 // Remove all bad triangles
990 triangulation.RemoveAll(t => badTriangles.Contains(t));
991
992 // Build new triangle for every edge of the polygon
993 foreach (var edge in polygon)
994 {
995 var newTriangle = new Triangle(edge.A,edge.B,point);
996 triangulation.Add(newTriangle);
997 }
998 }
999
1000 // Remove all triangles which contain a point of the super triangle
1001 triangulation.RemoveAll(t => t.ContainsPoints(superTriangle.Points));
1002
1003 return triangulation;

1011 private List<Edge> GenerateVoronoi(List<Triangle> triangulation)
1012 {
1013 var bisectors = new List<Edge>();
1014 var toRemove = new List<int>();
1015 var voronoi = new List<Edge>();
1016
1017 // ONLY FOR DEBUGGING
1018 _debugBisectors = new List<Edge>();
1019 _debugVoronoiIntersections = new List<Point>();
1020 // ONLY FOR DEBUGGING
1021
1022 // Generate three edges from every circumcenter to the edge of the triangle
1023 foreach (var triangle in triangulation)
1024 {
1025 var center = triangle.GetCircumcircle().Center;
1026
1027 // Only Centers within the map
1028 if (center.X < 0 || center.X > size || center.Y < 0 || center.Y > size)
1029 continue;
1030
1031 // Add bisectors to a list for later use
1032 foreach (var edge in triangle.Edges)
1033 {
1034 var eTmp = new Edge(center, new Point(((edge.A.X + edge.B.X) / 2), ((edge.A.Y + edge.B.Y) / 2)));
1035 eTmp.Room1 = edge.A;
1036 eTmp.Room2 = edge.B;
1037 bisectors.Add(eTmp);
1038
1039 // ONLY FOR DEBUGGING
1040 _debugBisectors.Add(new Edge(center, new Point(((edge.A.X + edge.B.X) / 2), ((edge.A.Y + edge.B.Y) / 2))));
1041 // ONLY FOR DEBUGGING
1042 }
1043 }
1044
1045 var edgeCount = bisectors.Count;
1046
1047 // Connect two bisectors that meet to one big edge
1048 for (var i = 0; i < edgeCount - 1; i++)
1049 {
1050 for (var j = i + 1; j < edgeCount; j++)
1051 {
1052 if (Point.Equals(bisectors[i].B, bisectors[j].B))
1053 {
1054 var longEdge = new Edge(bisectors[i].A, bisectors[j].A);
1055 longEdge.Room1 = bisectors[i].Room1;
1056 longEdge.Room2 = bisectors[i].Room2;
1057 longEdge.GiveID();
1058 voronoi.Add(longEdge);
1059 toRemove.Add(i);
1060 toRemove.Add(j);
1061 }
1062 }
1063 }
1064
1065 // Prevent out of bounds exception
1066 var offset = 0;
1067 toRemove.Sort();
1068
1069 // Remove all bisectors that have been connected
1070 foreach (var count in toRemove)
1071 {
1072 bisectors.RemoveAt(count - offset);
1073 offset++;
1074 }
1075
1076 // For each remaining bisector, so every bisector that goes to the outside of the dungeon
1077 foreach (var e in bisectors)
1078 {
1079 // Finde das eine Dreieck, dem dieser Bisektor entstammt
1080 Triangle owner = null;
1081 foreach (var tri in triangulation)
1082 {
1083 if (Point.Equals(tri.GetCircumcircle().Center, e.A))
1084 {
1085 owner = tri;
1086 break;
1087 }
1088 }
1089 if (owner == null) continue;
1090
1091 // direction bisector
1092 var dir = new Vector2(e.B.X - e.A.X, e.B.Y - e.A.Y);
1093 dir.Normalize();
1094
1095 // Checkpoints slightly forward and backward
1096 var forwardTest = new Point(e.B.X + dir.x * 0.01f, e.B.Y + dir.y * 0.01f);
1097 var forwardInside = !PointOutTriangle(forwardTest, owner.Points[0], owner.Points[1], owner.Points[2]);
1098
1099 // If the forward point is inside,
1100 // we have to extend backwards outwards, otherwise forwards.
1101 var rayDir = forwardInside ? new Vector2(-dir.x, -dir.y) : dir;
1102
1103 // Build a ray from the circumcenter far outwards
1104 var far = new Point(e.A.X + rayDir.x * size * 2f, e.A.Y + rayDir.y * size * 2f);
1105 var ray = new Edge(e.A, far);
1106
1107 // Intersection with the map boundary
1108 var inter = FindIntersectionWithMapBoundary(ray, size);
1109 _debugVoronoiIntersections.Add(inter);
1110
1111 // New Voronoi-Edge
1112 var newEdge = new Edge(e.A, inter);
1113 newEdge.Room1 = e.Room1;
1114 newEdge.Room2 = e.Room2;
1115 newEdge.GiveID();
1116 voronoi.Add(newEdge);
1117 }
1118 // Return the voronoi diagram as a list of Edges
1119 return voronoi;
1120 }

1132 private bool PointOutTriangle(Point pt, Point v1, Point v2, Point v3)
1133 {
1134 var d1 = Sign(pt, v1, v2);
1135 var d2 = Sign(pt, v2, v3);
1136 var d3 = Sign(pt, v3, v1);
1137
1138 var hasNeg = (d1 < 0) || (d2 < 0) || (d3 < 0);
1139 var hasPos = (d1 > 0) || (d2 > 0) || (d3 > 0);
1140
1141 return (hasNeg && hasPos);
1142 }

1151 private float Sign(Point p1, Point p2, Point p3)
1152 {
1153 return (p1.X - p3.X) * (p2.Y - p3.Y) - (p2.X - p3.X) * (p1.Y - p3.Y);

1162 private Point FindIntersectionWithMapBoundary(Edge edge, float dungeonSize)
1163 {
1164 var dx = edge.B.X - edge.A.X;
1165 var dy = edge.B.Y - edge.A.Y;
1166
1167 var vertical = Mathf.Abs(dx) < 1e-5f;
1168 var horizontal = Mathf.Abs(dy) < 1e-5f;
1169
1170 var candidates = new List<Point>();
1171
1172 if (vertical)
1173 {
1174 candidates.Add(new Point(edge.B.X, 0));
1175 candidates.Add(new Point(edge.B.X, dungeonSize));
1176 }
1177 else if (horizontal)
1178 {
1179 candidates.Add(new Point(0, edge.B.Y));
1180 candidates.Add(new Point(dungeonSize, edge.B.Y));
1181 }
1182 else
1183 {
1184 var m = dy / dx;
1185 var b = edge.B.Y - m * edge.B.X;
1186
1187 candidates.Add(new Point(0, b)); // left
1188 candidates.Add(new Point(dungeonSize, m * dungeonSize + b)); // right
1189 candidates.Add(new Point((0 - b) / m, 0)); // bottom
1190 candidates.Add(new Point((dungeonSize - b) / m, dungeonSize)); // top
1191 }
1192
1193 // Only valid points in Range [0, size]
1194 candidates.RemoveAll(p => p.X < 0 || p.X > dungeonSize || p.Y < 0 || p.Y > dungeonSize);
1195
1196 return CheckPoints(candidates, edge.B);
1197 }

1205 private Point CheckPoints(List<Point> points, Point reference)
1206 {
1207 var result= new Point(0,0);
1208 float minDistance = 10000;
1209 foreach (var p in points)
1210 {
1211 var distx = reference.X - p.X;
1212 var disty = reference.Y - p.Y;
1213 var distance = Mathf.Sqrt((distx*distx)+(disty*disty));
1214 if (distance < minDistance)
1215 {
1216 minDistance = distance;
1217 result = p;
1218 }
1219 }
1220 return result;
1221 }
