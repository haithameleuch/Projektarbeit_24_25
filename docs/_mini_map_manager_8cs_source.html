165 private void GenerateRoomIcons()
166 {
167 // clear existing icons except playerIcon
168 foreach (Transform child in roomsContainer)
169 if (child != playerIcon) Destroy(child.gameObject);
170
171 _roomIcons.Clear();
172 _roomImages.Clear();
173 _roomLabels.Clear();
174
175 foreach (var room in _dungeon.Rooms)
176 {
177 var go = Instantiate(roomIconPrefab, roomsContainer);
178 var rt = go.GetComponent<RectTransform>();
179 var img = go.GetComponent<Image>();
180 var label = go.GetComponentInChildren<TMP_Text>();
181
182 // map world coords (0–dungeonSize) to UI coords (0–minimap width/height)
183 var x = (room.Center.X / _dungeonSize) * minimapRect.rect.width;
184 var y = (room.Center.Y / _dungeonSize) * minimapRect.rect.height;
185 rt.anchoredPosition = new Vector2(x, y);
186
187 var visited = room.Visited;
188 img.color = visited ? visitedColor : unvisitedColor;
189 if (label is not null)
190 {
191 label.enabled = visited;
192 label.text = visited ? room.Type.ToString() : string.Empty;
193 }
194
195 _roomIcons[room.ID] = rt;
196 _roomImages[room.ID] = img;
197 if (label is not null) _roomLabels[room.ID] = label;
198 }
199 }

204 private void GenerateConnections()
205 {
206 _connectionLines.ForEach(Destroy);
207 _connectionLines.Clear();
208
209 foreach (var room in _dungeon.Rooms)
210 {
211 foreach (var neighbor in room.Neighbors)
212 {
213 if (room.ID >= neighbor.ID) continue;
214
215 var a = _roomIcons[room.ID].anchoredPosition;
216 var b = _roomIcons[neighbor.ID].anchoredPosition;
217 var diff = b - a;
218
219 var distance = diff.magnitude;
220 var angle = Mathf.Atan2(diff.y, diff.x) * Mathf.Rad2Deg;
221
222 var line = Instantiate(linePrefab, connectionsContainer);
223 var lrt = line.GetComponent<RectTransform>();
224 lrt.sizeDelta = new Vector2(distance, lineThickness);
225 lrt.anchoredPosition = a + lineOffset;
226 lrt.localRotation = Quaternion.Euler(0, 0, angle);
227
228 _connectionLines.Add(line);
229 }
230 }
