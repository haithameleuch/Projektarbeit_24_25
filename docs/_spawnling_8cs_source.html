218 public override void OnActionReceived(ActionBuffers actions)

223 var turnInput = actions.ContinuousActions[1]; // Left/right turn
224
225 // Move forward/backward
226 var forwardMovement = transform.forward * moveInput * movementSpeed * Time.deltaTime;
227 _rb.MovePosition(_rb.position + forwardMovement);
228
229 // Rotate
230 transform.Rotate(Vector3.up, turnInput * 180f * Time.deltaTime);
231
232 // Calculate distance-based progress reward
233 var currentDistance = Vector3.Distance(transform.localPosition, target.transform.localPosition);
234 var distanceDelta = _prevDistance - currentDistance;
235 AddReward(distanceDelta * 0.01f);
236 _prevDistance = currentDistance;

238 // Reward for facing the target
239 var toTarget = (target.transform.localPosition - transform.localPosition).normalized;
240 var facingDot = Vector3.Dot(transform.forward, toTarget); // 1 = facing directly at target
241 AddReward(facingDot * 0.005f);
242
243 // Step penalty to encourage faster completion
244 AddReward(-0.001f);
