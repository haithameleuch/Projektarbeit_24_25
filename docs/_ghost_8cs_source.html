183 public override void OnActionReceived(ActionBuffers actions)
184 {
185 if (!target) { AddReward(-0.001f); return; }
186
187 var moveInput = actions.ContinuousActions[0]; // Forward/backward movement input
188 var turnInput = actions.ContinuousActions[1]; // Rotation input
189
190 // Move forward/backward based on input and speed
191 var forwardMovement = transform.forward * moveInput * movementSpeed * Time.deltaTime;
192 _rb.MovePosition(_rb.position + forwardMovement);
193
194 // Rotate around Y axis based on input
195 transform.Rotate(Vector3.up, turnInput * 180f * Time.deltaTime);
196
197 // Reward agent for reducing distance to target
198 var currentDistance = Vector3.Distance(transform.localPosition, target.transform.localPosition);
199 var distanceDelta = _prevDistance - currentDistance;
200 AddReward(distanceDelta * 0.01f);
201 _prevDistance = currentDistance;
202
203 // Small reward for facing the target
204 var toTarget = (target.transform.localPosition - transform.localPosition).normalized;
205 var facingDot = Vector3.Dot(transform.forward, toTarget);
206 AddReward(facingDot * 0.005f);
207
208 AddReward(-0.001f); // Small time penalty to encourage efficiency
209
210 // Detect if stuck by checking minimal movement
211 if (Vector3.Distance(transform.localPosition, _lastPosition) < 0.01f)
212 {
213 _stuckTimer += Time.deltaTime;
214 if (_stuckTimer > 2f)
215 {
216 AddReward(-5f); // Penalize for being stuck
217 EndEpisode();
218 }
219 }
220 else
221 {
222 _stuckTimer = 0f;
223 }

251 private IEnumerator FadeToAlpha(float targetAlpha)
