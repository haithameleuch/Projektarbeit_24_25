231 private void CalculatePixel()
232 {
233 _xPixel = 0;
234 _yPixel = 0;
235
236 // Get the Cinemachine Virtual Camera (used for raycasting)
237 var cinemachineCamera = GameObject
238 .FindWithTag("CanvCamera")
239 ?.GetComponent<CinemachineCamera>();
240
241 if (cinemachineCamera == null) return;
242
243 // Get the main Unity Camera for raycasting
244 var mainCamera = Camera.main;
245 if (mainCamera is null) return;
246
247 // Cast a ray from the mouse position in 3D space
248 var ray = mainCamera.ScreenPointToRay(Input.mousePosition);
249
250 // Check if the ray hits any object (such as the canvas)
251 if (Physics.Raycast(ray, out var hit, Mathf.Infinity))
252 {
253 point.position = hit.point; // Move the pointer to the hit point
254
255 // Convert world point to local space of the canvas
256 var localPoint = transform.InverseTransformPoint(hit.point);
257
258 // Calculate pixel position from local space
259 var width = transform.localScale.x;
260 var height = transform.localScale.y;
261
262 var normalizedX = (localPoint.x + width * 0.5f) / width;
263 var normalizedY = (localPoint.y + height * 0.5f) / height;
264
265 _xPixel = (int)(normalizedX * totalXPixels);
266 _yPixel = (int)(normalizedY * totalYPixels);
267
268 ChangePixelsAroundPoint();
269 }
270 else
271 {
272 _pressedLastFrame = false;
273 }
274 }

279 private void ChangePixelsAroundPoint()
280 {
281 if (useInterpolation && _pressedLastFrame && (_lastX != _xPixel || _lastY != _yPixel))
282 {
283 // Calculate the distance between the current and previous pixel positions
284 var dist = (int)
285 Mathf.Sqrt(
286 (_xPixel - _lastX) * (_xPixel - _lastX) + (_yPixel - _lastY) * (_yPixel - _lastY)
287 );
288
289 // Interpolate between the two points and apply brush to each intermediate pixel
290 for (var i = 1; i <= dist; i++)
291 {
292 var interpolatedX = (i * _xPixel + (dist - i) * _lastX) / dist;
293 var interpolatedY = (i * _yPixel + (dist - i) * _lastY) / dist;
294 DrawBrush(interpolatedX, interpolatedY);
295 }
296 }
297 else
298 {
299 DrawBrush(_xPixel, _yPixel);
300 }
301
302 // Update state for next frame
303 _pressedLastFrame = true;
304 _lastX = _xPixel;
305 _lastY = _yPixel;
306 SetTexture();
307 }

314 private void DrawBrush(int xPix, int yPix)
315 {
316 var i = Mathf.Max(0, xPix - brushSize + 1);
317 var j = Mathf.Max(0, yPix - brushSize + 1);
318 var maxi = Mathf.Min(totalXPixels - 1, xPix + brushSize - 1);
319 var maxj = Mathf.Min(totalYPixels - 1, yPix + brushSize - 1);
320
321 // Loop over the brush area and apply color to each pixel within the brush's radius
322 for (var x = i; x <= maxi; x++)
323 {
324 for (var y = j; y <= maxj; y++)
325 {
326 if ((x - xPix) * (x - xPix) + (y - yPix) * (y - yPix) <= brushSize * brushSize)
327 {
328 _colorMap[y * totalYPixels + x] = _brushColor;
329 }
330 }
331 }
332 }

337 private void SetTexture()
338 {
339 // Create a rotated copy of the color map (for visual effect or canvas alignment)
340 var rotatedColorMap = new Color[_colorMap.Length];
341 for (var y = 0; y < totalYPixels; y++)
342 {
343 for (var x = 0; x < totalXPixels; x++)
344 {
345 var originalIndex = y * totalXPixels + x;
346 var rotatedIndex = (totalYPixels - 1 - y) * totalXPixels + (totalXPixels - 1 - x);
347 rotatedColorMap[rotatedIndex] = _colorMap[originalIndex];
348 }
349 }
350
351 // Assign the rotated color map to the texture and apply it
352 generatedTexture.SetPixels(rotatedColorMap);
353 generatedTexture.Apply();
354 }

375 private Texture2D Preprocessing(Texture2D texture2D, int newWidth, int newHeight)
376 {
377 var rt = new RenderTexture(newWidth, newHeight, 24)
378 {
379 filterMode = FilterMode.Bilinear
380 };
381 RenderTexture.active = rt;
382
383 // Blit the original texture into the render texture
384 Graphics.Blit(texture2D, rt);
385
386 // Read the render texture into a new texture
387 var result = new Texture2D(newWidth, newHeight, TextureFormat.RGB24, false);
388 result.ReadPixels(new Rect(0, 0, newWidth, newHeight), 0, 0);
389
390 // Flip the pixels vertically and horizontally, then invert the colors
391 var pixels = result.GetPixels();
392 var flippedPixels = new Color[pixels.Length];
393 var rowLength = newWidth;
394
395 for (var y = 0; y < newHeight; y++)
396 {
397 for (var x = 0; x < newWidth; x++)
398 {
399 // Calculate the flipped index
400 var flippedIndex = (newWidth - 1 - x) + (newHeight - 1 - y) * rowLength;
401
402 // Invert the color
403 var originalColor = pixels[x + y * rowLength];
404 Color invertedColor;
405 if (glyph)
406 {
407 invertedColor = new Color(
408 originalColor.r,
409 originalColor.g,
410 originalColor.b,
411 originalColor.a
412 );
413
414 }
415 else
416 {
417 invertedColor = new Color(
418 1- originalColor.r,
419 1- originalColor.g,
420 1- originalColor.b,
421 originalColor.a
422 );
423 }
424
425 flippedPixels[flippedIndex] = invertedColor;
426 }
427 }
428 result.SetPixels(flippedPixels);
429 result.Apply();
430
431 // Clean up the render texture
432 RenderTexture.active = null;
433 rt.Release();
434
435 return result;
436 }

512 private void ValidGlyph(int glyphDigit)
513 {
514 // If digit is not in the list invalid guess
515 if (!_refGlyph.Contains(glyphDigit)) return;
516
517 // Remove the correct glyph from the list
518 _refGlyph.Remove(glyphDigit);
519
520 // remove the glyph from inventory
521 var player = GameObject.FindWithTag("Player");
522 var inventory = player.GetComponent<Inventory.Inventory>();
523
524 var itemName = _digitToString[glyphDigit];
525
526 if (!string.IsNullOrEmpty(itemName))
527 {
528 var removeItemIndex = inventory.GetItemByName(CapitalizeFirstLetter(itemName));
529 inventory.RemoveItem(removeItemIndex.Item1, removeItemIndex.Item2);
530 }
531
532 // If all glyphs have been guessed spawn boss key
533 if (_refGlyph.Count != 0) return;
534 if (spawnPrefab)
535 {
536 if (SaveSystemManager.SaveData != null)
537 SaveSystemManager.SaveData.GlyphMiniGameCleared = true;
538
539 // Spawn near canvas position
540 var spawnPos = transform.TransformPoint(spawnPrefabOffset);
541 Instantiate(spawnPrefab, spawnPos, Quaternion.identity);
542 Debug.Log("You have the boss key!");
543 }

545 {
