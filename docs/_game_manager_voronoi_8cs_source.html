245 private static List<string> GetGlyphItem(List<int> glyphKeys)
246 {
247 // Map of keys to glyph names
248 var keyMapGlyph = new Dictionary<int, string>
249 {
250 { 0, "Air" },
251 { 1, "Earth" },
252 { 2, "Energy" },
253 { 3, "Fire" },
254 { 4, "Power" },
255 { 5, "Power" },
256 { 6, "Time" },
257 { 7, "Water" },
258 };
259
260 // Get the glyph names for the given keys
261 var glyphNames = glyphKeys
262 .Where(key => keyMapGlyph.ContainsKey(key)) // ensures only valid keys
263 .Select(key => keyMapGlyph[key])
264 .ToList();
265
266 return glyphNames;
267 }

274 private static List<int> GenerateRandomGlyphs(int seed = -1)
275 {
276 var setRefGlyphKeys = new List<int>();
277 // Create Random with optional seed
278 var rand = seed >= 0 ? new System.Random(seed) : new System.Random();
279 // Decide how many additional numbers to add (2 to 4)
280 var x = rand.Next(2, 5);
281
282 // candidates: 0-7 excluding 5. since it repeated two times in glyphs
283 var candidates = Enumerable.Range(0, 8).Where(n => n != 5 && n != 4).ToList();
284 // Shuffle candidates
285 candidates = candidates.OrderBy(n => rand.Next()).ToList();
286 // Take x numbers and add
287 setRefGlyphKeys.AddRange(candidates.Take(x));
288
289 return setRefGlyphKeys;
290 }

374 private void TrackCurrentRoom()
375 {
376 var playerPos = new Vector2(_player.transform.position.x, _player.transform.position.z);
377 var minDist = Vector2.Distance(playerPos, new Vector2(_currentRoom.Center.X, _currentRoom.Center.Y));
378
379 if (minDist <= roomSwitchThreshold)
380 return;
381
382 var closest = _currentRoom;
383
384 foreach (var neighbor in _currentRoom.Neighbors)
385 {
386 var dist = Vector2.Distance(playerPos, new Vector2(neighbor.Center.X, neighbor.Center.Y));
387 if (dist < minDist)
388 {
389 minDist = dist;
390 closest = neighbor;
391 }
392 }
393
394 if (closest != _currentRoom)
395 {
396 _currentRoom = closest;
397 OnRoomEntered(_currentRoom);
398 }
399 }

406 private void OnRoomEntered(Room newRoom)
407 {
408 Debug.Log($"[GameManager] Player entered room {newRoom.ID} (Type: {newRoom.Type})");
409
410 if (newRoom.Visited) return;
411 newRoom.Visited = true;
412
413 SaveSystemManager.SetRoomVisited(newRoom.ID, true);
414 SaveSystemManager.SetCurrentRoomID(newRoom.ID);
415
416 switch (newRoom.Type)
417 {
418 case RoomType.Start:
419 break;
420 case RoomType.Normal:
421 break;
422 case RoomType.Item:
423 break;
424 case RoomType.MiniGame:
425 //EventManager.Instance.TriggerCloseDoors();
426 break;
427 case RoomType.Enemy:
428 _enemySpawner.ActivateEnemyInRoom(newRoom);
429 EventManager.Instance.TriggerCloseDoors();
430 break;
431 case RoomType.Boss:
432 _bossSpawner?.ActivateBossInRoom(newRoom);
433 EventManager.Instance.TriggerCloseBossDoors();
434 break;
435 default:
436 Debug.Log("Unknown room type");
437 break;
438 }
439 }

457 public bool OnBossKeyUsed()
458 {
459 var bossRoom = _dungeon.GetBossRoom();
460
461 if (_currentRoom != null && bossRoom != null && _currentRoom.Neighbors.Contains(bossRoom))
462 {
463 Debug.Log("[GameManager] Boss doors are opened!");
464 EventManager.Instance.TriggerOpenBossDoors();
465
466 SaveSystemManager.SetBossRoomOpen(true);
467 return true;
468 }
469
470 Debug.Log("[GameManager] Boss key used, but not in front of the boss room.");
471 return false;
472 }
