160 public override void CollectObservations(VectorSensor sensor)
161 {
162 if (!target || !shootPoint)
163 {
164 sensor.AddObservation(0f); // Normalized health
165 sensor.AddObservation(Vector3.zero); // Local direction to target
166 sensor.AddObservation(0f); // Alignment score
167 sensor.AddObservation(1f); // Angle difference
168 return;

170
171 // Calculate normalized health of the target
172 var targetStats = target.GetComponent<Stats.Stats>();
173 if (targetStats != null)
174 {
175 var targetMaxHealth = targetStats.GetMaxStats(0);
176 var targetCurHealth = targetStats.GetCurStats(0);
177 _healthNormalized = targetCurHealth / targetMaxHealth;
178 }
179 else
180 {
181 _healthNormalized = 0f;
182 }
183
184 // Direction to target in world space
185 var toTarget = (target.transform.position - shootPoint.position).normalized;
186 var forward = shootPoint.forward;
187
188 // Observations:
189 sensor.AddObservation(_healthNormalized); // Target health [0,1] - 1 Observation
190 sensor.AddObservation(shootPoint.InverseTransformDirection(toTarget)); // Direction to target in local space - 3 Observations
191 sensor.AddObservation(Vector3.Dot(forward, toTarget)); // Alignment measure (dot product) - 1 Observation
192 sensor.AddObservation(Vector3.Angle(forward, toTarget) / 180f); // Angular difference normalized - 1 Observation
193 }
