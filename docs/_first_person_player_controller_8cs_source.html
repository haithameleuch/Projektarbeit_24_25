121 private void HandleMovement(Vector3 moveDir)
122 {
123 var speed = moveSpeed;
124 if (_stats != null)
125 {
126 var curList = _stats.GetCurStatsList();
127 if (speedStatIndex >= 0 && speedStatIndex < curList.Count)
128 speed = _stats.GetCurStats(speedStatIndex);
129 }
130
131 var moveDistance = speed * Time.deltaTime; // Maximum distance the player can move this frame.
132 const float playerRadius = 0.85f; // Radius of the player's capsule for collision detection.
133 const float playerHeight = 2f; // Height of the player's capsule for collision detection.
134
135 // Check for collisions in the movement direction using CapsuleCast.
136 if (
137 Physics.CapsuleCast(
138 transform.position,
139 transform.position + Vector3.up * playerHeight,
140 playerRadius,
141 moveDir,
142 out RaycastHit hit,
143 moveDistance
144 )
145 )
146 {
147 // If a collision is detected, calculate the slide direction along the wall.
148 if (hit.collider)
149 {
150 Vector3 slideDir = Vector3.ProjectOnPlane(moveDir, hit.normal);
151
152 // Perform a secondary check to prevent sliding at sharp corners.
153 if (
154 Physics.CapsuleCast(
155 transform.position,
156 transform.position + Vector3.up * playerHeight,
157 playerRadius,
158 slideDir,
159 out var edgeHit,
160 moveDistance
161 )
162 )
163 {
164 slideDir = Vector3.zero; // Stop movement at sharp corners.
165 }
166
167 transform.position += slideDir * moveDistance;
168 }
169 }
170 else
171 {
172 // If no collision is detected, move the player normally.
173 transform.position += moveDir * moveDistance;
174 }
175
176 var pos = transform.position;
177 pos.y = 1f;
178 transform.position = pos;

184 private void HandleRotation()
185 {
186 // Get mouse-look input.
187 var lookDelta = gameInput.GetLookDelta();
188 _rotationY += lookDelta.x * mouseSensitivity;
189 if (lookDelta == Vector2.zero)
190 return;
191
192 // Calculate the target Yaw rotation based on mouse input.
193 var targetYaw = Quaternion.Euler(0f, _rotationY, 0f);
194
195 // Smoothly interpolate to the targetYaw rotation using Slerp.
196 _currentRotation = Quaternion.Slerp(_currentRotation, targetYaw, rotationSmoothness);
197 transform.rotation = _currentRotation;
198
199 if (allowPitchRotation)
200 {
201 // Get the mouse-look input.
202 _rotationX -= lookDelta.y * mouseSensitivity;
203 _rotationX = Mathf.Clamp(_rotationX, -90f, 90f);
204
205 // Calculate the target Pitch rotation based on mouse input.
206 var targetPitch = Quaternion.Euler(_rotationX, 0f, 0f);
207
208 // Smoothly interpolate to the targetPitch rotation using Slerp.
209 _currentCameraRotation = Quaternion.Slerp(_currentCameraRotation, targetPitch, rotationSmoothness);
210 cameraTransform.localRotation = _currentCameraRotation;
211 }
